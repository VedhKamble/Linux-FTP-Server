# To test ftp server connection and access from client machine you must know # # from what username you are accessing it because that user must be present in # that server machine 

# First for Anonymous access

ftp <server machine name>
# You will be prompted to enter username so enter
ftp 
# Or you can also use anonymous in place of ftp
# Then passwd will be prompted 
# Here you can enter anything or nothing and just press Enter
# You're now logged-in as anonymous user

# Second for Real User 

ftp <server machine name>
# Enter the VLU name created on server machine 
# Next enter the passwd used for it
# You're now logged-in as Real user

# Third for Chroot User

ftp <server machine name>
# Enter the chroot user name 
# Also the passwd used while creating it
# You're now logged-in as chroot user
