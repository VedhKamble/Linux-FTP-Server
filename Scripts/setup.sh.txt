#!/bin/bash

# This file contents setup instructions for ftp server

# Check whether you have ftp server package named vsftpd installed on your machine

systemctl status vsftpd.service

# If it shows "service not found" then install it with yum/dnf

# Before installing check whether your system contains the rpm packages of ftp server

rpm -qa | grep vsftpd*

# Once they are visible start installing them

dnf install vsftp* -y

# Upon installation start the ftp server

systemctl enable vsftpd

systemctl start vsftpd

# And then check the service status

systemctl status vsftpd

# Now create the VLUs (Valid Linux Users) who are going to access this server from client machine

useradd hr
passwd hr

useradd sales1
passwd sales1

useradd sales2
passwd sales2

useradd intern1
passwd intern1

useradd intern2
passwd intern2

useradd ram
passwd ram

# Now create couple of files inside each of the user /home dir for testing purpose

cd /home/hr
touch h1 h2

# Likewise do it for rest of the users

# Now let us configure the conf file of ftp server i.e., vsftpd.conf

vi /etc/vsftpd/vsftpd.conf

# Now modify the following directives of this file 

anonymous_enable=YES
local_enable=YES
write_enable=YES
local_umask=022
dirmessage_enable=YES
xferlog_enable=YES
connect_from_port_20=YES
xferlog_file=/var/log/xferlog
xferlog_std_format=YES
idle_session_timeout=600
ftpd_banner=Welcome to Vedh's FTP Server.
chroot_list_enable=YES
allow_writeable_chroot=YES
chroot_list_file=/etc/vsftpd/chroot_list
listen=YES
listen_ipv6=NO
pam_service_name=vsftpd
userlist_enable=YES
userlist_file=/etc/vsftpd/user_list
userlist_deny=YES

# Save & close the file 

# Now add user "ram" to user_list & ftpuser file

cd /etc/vsftpd
vi user_list
# Add user "ram" at the end of the list and save & close it, similarly
cd /etc/vsftpd
vi ftpusers
# Add user "ram" at the end of the list and save & close it

# Now add users "intern1" & "intern2" in chroot_list for that create a chroot_list file

cd /etc/vsftpd
vi chroot_list
# Add users "intern1" & "intern2"

# At last add ftp service to firewall & reload firewall

firewall-cmd --permanent --add-service=ftp
firewall-cmd --reload

# Also restart vsftpd service

systemctl restart vsftpd

# All set and you are good to go
